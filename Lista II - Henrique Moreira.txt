esq = int(input('Digite o valor do primeiro lado: '))
dit = int(input('Digite o valor do segundo lado: '))
base = int(input('Digite o valor da base: '))
 
if esq + dit < base:
    print('A soma dos dois lados não pode ser menor que a base!')
elif esq == dit:
    print('Triângulo isósceles')
elif esq == dit and esq == base:
    print('Triângulo equilátero')
elif esq != dit and esq != base and dit != base:
    print('Triângulo escaleno')
